## DDL

```sql
-- Smoke-only: sys_menu, sys_version, imbs_permission
-- Target: PostgreSQL 16+

CREATE TABLE IF NOT EXISTS sys_version (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    version_code VARCHAR(50) NOT NULL UNIQUE,
    version_name VARCHAR(100) NOT NULL,
    description VARCHAR(500),
    icon VARCHAR(50),
    sort_order INTEGER DEFAULT 0,
    is_active SMALLINT DEFAULT 1,
    is_default SMALLINT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS sys_menu (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    parent_id BIGINT DEFAULT 0,
    menu_path VARCHAR(200) NOT NULL UNIQUE,
    menu_title VARCHAR(100) NOT NULL,
    menu_icon VARCHAR(50),
    menu_type VARCHAR(20) DEFAULT 'menu',
    has_children SMALLINT DEFAULT 0,
    redirect_path VARCHAR(200),
    sort_order INTEGER DEFAULT 0,
    is_visible SMALLINT DEFAULT 1,
    remark VARCHAR(255),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS imbs_permission (
    id VARCHAR(32) PRIMARY KEY,
    perm_code VARCHAR(64) NOT NULL UNIQUE,
    description VARCHAR(128) NOT NULL,
    extra JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Minimal seed rows (smoke-only)

INSERT INTO sys_version (version_code, version_name, description, is_active, is_default)
SELECT 'local-smoke', 'Local Smoke Version', 'PostgreSQL local smoke version', 1, 1
WHERE NOT EXISTS (SELECT 1 FROM sys_version WHERE version_code = 'local-smoke');

INSERT INTO sys_menu (menu_path, menu_title, parent_id, sort_order, is_visible)
SELECT '/system', 'System', 0, 10, 1
WHERE NOT EXISTS (SELECT 1 FROM sys_menu WHERE menu_path = '/system');

INSERT INTO sys_menu (menu_path, menu_title, parent_id, sort_order, is_visible)
SELECT '/system/permissions', 'Permission Management', 0, 20, 1
WHERE NOT EXISTS (SELECT 1 FROM sys_menu WHERE menu_path = '/system/permissions');

INSERT INTO imbs_permission (id, perm_code, description)
SELECT 'a1b2c3d4e5f6478990abcdef12345678', 'system:read', 'Local smoke permission for system read APIs'
WHERE NOT EXISTS (SELECT 1 FROM imbs_permission WHERE perm_code = 'system:read');
```
